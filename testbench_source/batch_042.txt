module xor_gate(
    input a,
    input b,
    output c
);
    wire w1, w2, w3;
    
    assign w1 = ~(a & b);
    assign w2 = a & ~b;
    assign w3 = ~a & b;
    
    assign c = w1 | w2 | w3;
endmodule

module pipeline_accumulator(
    input clk,
    input rst_n,
    input [7:0] data_in,
    input valid_in,
    output reg [9:0] data_out,
    output reg valid_out,
    output reg ready_in
);
    reg [9:0] accumulator;
    reg [2:0] count;
    
    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            accumulator <= 0;
            count <= 0;
            valid_out <= 0;
            ready_in <= 1;
        end
        else begin
            if (valid_in) begin
                accumulator <= accumulator + data_in;
                count <= count + 1;
                ready_in <= 0;
                
                if (count == 7) begin
                    data_out <= accumulator;
                    valid_out <= 1;
                    accumulator <= 0;
                    count <= 0;
                end
            end
            else begin
                ready_in <= 1;
            end
        end
    end
endmodule

module final_module(
    input a,
    input b,
    input clk,
    input rst_n,
    input [7:0] data_in,
    input valid_a,
    input ready_b,
    output ready_a,
    output reg valid_b,
    output reg [9:0] data_out
);
    wire xor_output;
    wire [9:0] accumulator_output;
    wire valid_out;
    wire ready_in;
    
    xor_gate xor_inst(
        .a(a),
        .b(b),
        .c(xor_output)
    );
    
    pipeline_accumulator accumulator_inst(
        .clk(clk),
        .rst_n(rst_n),
        .data_in(data_in),
        .valid_in(valid_a),
        .data_out(accumulator_output),
        .valid_out(valid_out),
        .ready_in(ready_in)
    );
    
    assign ready_a = ready_in & ready_b;
    
    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            valid_b <= 0;
            data_out <= 0;
        end
        else begin
            if (valid_out & valid_b) begin
                data_out <= accumulator_output + data_out;
                valid_b <= 0;
            end
            else if (valid_out & ready_a) begin
                data_out <= accumulator_output;
                valid_b <= 1;
            end
        end
    end
endmodule

module nios_system_alu_carry_out (
  // inputs:
  address,
  clk,
  in_port,
  reset_n,

  // outputs:
  readdata
);

  output  [31:0] readdata;
  input   [1:0]  address;
  input          clk;
  input          in_port;
  input          reset_n;

  wire    clk_en;
  wire    data_in;
  wire    read_mux_out;
  reg     [31:0] readdata;
  assign  clk_en = 1;

  //s1, which is an e_avalon_slave
  assign read_mux_out = {1 {(address == 2'b00)}} & (in_port & data_in);
  assign read_mux_out = read_mux_out | {1 {(address == 2'b01)}} & (in_port | data_in);

  always @(posedge clk or negedge reset_n) begin
    if (~reset_n) begin
      readdata <= 0;
    end else if (clk_en) begin
      readdata <= {32'b0 | read_mux_out};
    end
  end

  assign data_in = in_port;

endmodule

module fifo_buffer(
  input rst,
  input wr_clk,
  input rd_clk,
  input [31:0]din,
  input wr_en,
  input rd_en,
  output reg [31:0]dout,
  output reg full,
  output reg empty
);

  reg [31:0] buffer [0:7];
  reg [2:0] wr_ptr;
  reg [2:0] rd_ptr;
  reg [2:0] count;

  always @(posedge wr_clk or negedge rst) begin
    if (rst) begin
      wr_ptr <= 3'b0;
      rd_ptr <= 3'b0;
      count <= 3'b0;
      full <= 1'b0;
      empty <= 1'b1;
    end else if (wr_en && !full) begin
      buffer[wr_ptr] <= din;
      wr_ptr <= wr_ptr + 1;
      count <= count + 1;
      if (count == 8) full <= 1'b1;
      empty <= 1'b0;
    end
  end

  always @(posedge rd_clk or negedge rst) begin
    if (rst) begin
      wr_ptr <= 3'b0;
      rd_ptr <= 3'b0;
      count <= 3'b0;
      full <= 1'b0;
      empty <= 1'b1;
    end else if (rd_en && !empty) begin
      dout <= buffer[rd_ptr];
      rd_ptr <= rd_ptr + 1;
      count <= count - 1;
      if (count == 1) empty <= 1'b1;
      full <= 1'b0;
    end
  end

endmodule
