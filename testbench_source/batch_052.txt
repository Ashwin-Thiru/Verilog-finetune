module ring_counter (
    input clk,
    input enable,
    input reset,
    output reg [3:0] q
);

reg [3:0] q_reg; // register to hold current value of counter

always @(posedge clk) begin
    if (reset) begin
        q_reg <= 4'b0000; // reset counter to 0000
    end else if (enable) begin
        q_reg <= {q_reg[2:0], q_reg[3]}; // shift counter value left by 1 bit
        q <= q_reg; // output shifted value
    end
end

endmodule

module barrel_shifter (
    input clk,
    input [15:0] DIN,
    input [3:0] SHIFT,
    output reg [15:0] DOUT
);

reg [15:0] stage1_out, stage2_out, stage3_out;

always @(posedge clk) begin
    stage1_out <= DIN << SHIFT;
end

always @(posedge clk) begin
    stage2_out <= stage1_out << 2;
end

always @(posedge clk) begin
    if (SHIFT[1:0] == 2'b00) begin
        stage3_out <= stage2_out;
    end else if (SHIFT[1:0] == 2'b01) begin
        stage3_out <= {stage2_out[13:0], stage2_out[15:14]};
    end else if (SHIFT[1:0] == 2'b10) begin
        stage3_out <= {stage2_out[14:0], stage2_out[15]};
    end else begin
        stage3_out <= {stage2_out[15], stage2_out[14:1]};
    end
end

always @(posedge clk) begin
    if (SHIFT[3] == 1) begin
        DOUT <= stage3_out >> SHIFT[2:0];
    end else begin
        DOUT <= stage3_out;
    end
end

endmodule
