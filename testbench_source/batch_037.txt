module PLL_DLL (
  input clk_in,
  input rst,
  input enable,
  input pll_freq,
  input dll_delay,
  output pll_out,
  output dll_out
);

  // Define any necessary parameters or constants here.
  parameter CLK_DIV = 2;
  parameter PHASE_BITS = 8;
  parameter CHARGE_PUMP_GAIN = 1;
  parameter MAX_DELAY = 10;
  parameter DELAY_BITS = 4;
  
  // Define internal signals
  reg [PHASE_BITS-1:0] phase_accumulator;
  reg [PHASE_BITS-1:0] delay_accumulator;
  reg [DELAY_BITS-1:0] delay_value;
  reg [PHASE_BITS-1:0] phase_error;
  reg [PHASE_BITS-1:0] delay_error;
  reg [PHASE_BITS-1:0] vco_control;
  reg [PHASE_BITS-1:0] charge_pump_output;
  reg [PHASE_BITS-1:0] delay_line_output;
  reg [PHASE_BITS-1:0] pll_output;
  reg [PHASE_BITS-1:0] dll_output;
  
  // Define PLL components
  always @(posedge clk_in or posedge rst) begin
    if (rst) begin
      phase_accumulator <= 0;
      vco_control <= 0;
    end else if (enable) begin
      phase_accumulator <= phase_accumulator + vco_control;
      phase_error <= pll_freq - (clk_in * CLK_DIV) / (1 << PHASE_BITS);
      charge_pump_output <= phase_error * CHARGE_PUMP_GAIN;
      vco_control <= charge_pump_output;
    end
  end
  
  // Define DLL components
  always @(posedge clk_in or posedge rst) begin
    if (rst) begin
      delay_accumulator <= 0;
      delay_value <= 0;
    end else if (enable) begin
      delay_accumulator <= delay_accumulator + delay_value;
      delay_error <= dll_delay - (delay_accumulator >> (PHASE_BITS - DELAY_BITS));
      delay_value <= delay_error;
    end
  end
  
  // Define frequency divider for PLL output
  always @(posedge clk_in or posedge rst) begin
    if (rst) begin
      pll_output <= 0;
    end else if (enable) begin
      pll_output <= (pll_output + 1) % CLK_DIV;
    end
  end
  
  // Define delay line for DLL output
  always @(posedge clk_in or posedge rst) begin
    if (rst) begin
      delay_line_output <= 0;
    end else if (enable) begin
      delay_line_output <= delay_accumulator >> PHASE_BITS;
    end
  end
  
  // Define outputs
  assign pll_out = (pll_output == 0) ? 1 : 0;
  assign dll_out = delay_line_output;
  
endmodule

module resultcounter (resultID, newresult, done, reset, globalreset, clk);

    input[1:0] resultID; 
    input newresult; 
    output[1:0] done; 
    wire[1:0] done;
    input reset; 
    input globalreset; 
    input clk; 

    reg[3:0] count; 
    reg[1:0] curr; 

    assign done = (count == 0) ? curr : 2'b00 ;

    always @(posedge clk or posedge reset or posedge globalreset)
    begin
        if (globalreset)
        begin
            count <= 4'b1000;
            curr <= 2'b00;
        end
        else if (reset)
        begin
            count <= 4'b1000;
        end
        else if (newresult && resultID != 2'b00 && count != 0)
        begin
            count <= count - 1;
            curr <= resultID;
        end
        else if (count == 0)
        begin
            curr <= 2'b00;
        end
    end 
endmodule

module counter (
  input clk,
  input reset,
  input enable,
  output reg [7:0] count
);

  always @(posedge clk) begin
    if (reset) begin
      count <= 8'b0;
    end else if (enable) begin
      count <= count + 1;
    end
  end

endmodule
