module barrel_shifter (
    input [3:0] data_in,
    input [1:0] shift_amount,
    input direction_left,
    output reg [3:0] data_out
);

reg [3:0] shift_reg [1:3];

always @(*) begin
    shift_reg[1] = data_in;
    shift_reg[2] = shift_reg[1] << 1;
    shift_reg[3] = shift_reg[2] << 1;
    
    if (shift_amount == 2'b00) begin
        data_out = data_in;
    end else if (shift_amount == 2'b01) begin
        data_out = shift_reg[2];
    end else if (shift_amount == 2'b10) begin
        data_out = shift_reg[3];
    end else if (shift_amount == 2'b11) begin
        data_out = direction_left ? shift_reg[3] : shift_reg[1];
    end
end

endmodule

module barrel_shifter (
    input [3:0] A,
    input [3:0] B,
    output reg [3:0] Q
);

reg [3:0] stage1_out;
reg [3:0] stage2_out;

always @(*) begin
    stage1_out = (B[3]) ? {A[2:0], 1'b0} : {1'b0, A[3:1]};
end

always @(*) begin
    stage2_out = (B[2]) ? {stage1_out[1:0], 2'b00} : {2'b00, stage1_out[3:2]};
end

always @(*) begin
    Q = (B[1]) ? {stage2_out[0], stage2_out[3:1]} : {stage2_out[2:0], stage2_out[3]};
end

endmodule

module parity_check (
  input clk,
  input reset,
  input [7:0] data_in,
  output reg parity_out
);

reg [7:0] pipeline_reg1;
reg [7:0] pipeline_reg2;
reg [7:0] pipeline_reg3;

always @(posedge clk) begin
  if (reset) begin
    pipeline_reg1 <= 8'b0;
    pipeline_reg2 <= 8'b0;
    pipeline_reg3 <= 8'b0;
    parity_out <= 1'b0;
  end else begin
    pipeline_reg1 <= data_in;
    pipeline_reg2 <= pipeline_reg1;
    pipeline_reg3 <= pipeline_reg2;
    parity_out <= ~(pipeline_reg1 ^ pipeline_reg2 ^ pipeline_reg3) & 1'b1;
  end
end

endmodule
