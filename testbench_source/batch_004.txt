module moore_state_machine (
  input clk,
  input rst,
  input in1,
  input in2,
  output reg out1,
  output reg out2
);

  // Define your states and state variables here
  parameter S0 = 2'b00;
  parameter S1 = 2'b01;
  parameter S2 = 2'b10;
  
  reg [1:0] state;
  
  // Define your outputs as a function of the current state
  always @ (state) begin
    case (state)
      S0: begin
        out1 = 1;
        out2 = 0;
      end
      S1: begin
        out1 = 0;
        out2 = 1;
      end
      S2: begin
        out1 = 1;
        out2 = 1;
      end
    endcase
  end
  
  // Define the state transitions based on the inputs and the current state
  always @ (posedge clk, posedge rst) begin
    if (rst) begin
      state <= S0;
    end else begin
      case (state)
        S0: begin
          if (in1) begin
            state <= S1;
          end else if (in2) begin
            state <= S2;
          end
        end
        S1: begin
          if (in2) begin
            state <= S2;
          end else if (!in1) begin
            state <= S0;
          end
        end
        S2: begin
          if (!in1 && !in2) begin
            state <= S0;
          end
        end
      endcase
    end
  end
  
endmodule

module fsm_8bit_sequence_detection (
  input clk,
  input reset,
  input [7:0] data,
  output reg [2:0] position
);

  parameter S0 = 3'b000, S1 = 3'b001, S2 = 3'b010, S3 = 3'b011, S4 = 3'b100;
  reg [2:0] state, next_state;
  
  always @(posedge clk) begin
    if (reset) begin
      state <= S0;
      position <= 0;
    end
    else begin
      state <= next_state;
      position <= (state == S4) ? 1 : 0;
    end
  end
  
  always @(*) begin
    next_state = state;
    case (state)
      S0: if (data == 8'b10101010) next_state = S4; else next_state = S1;
      S1: if (data[6:0] == 7'b0101010) next_state = S3; else if (data == 8'b10101010) next_state = S4; else next_state = S1;
      S2: if (data[4:0] == 5'b10101) next_state = S3; else if (data == 8'b10101010) next_state = S4; else next_state = S1;
      S3: if (data == 8'b10101010) next_state = S4; else if (data[6:0] == 7'b0101010) next_state = S3; else if (data[4:0] == 5'b10101) next_state = S3; else next_state = S1;
      S4: next_state = S4;
    endcase
  end
  
endmodule

module RefModule (
  input clk,
  input reset,
  input data,
  output reg shift_ena,
  output reg counting,
  input done_counting,
  output reg done,
  input ack
);

  typedef enum logic[3:0] {
    S, S1, S11, S110, B0, B1, B2, B3, Count, Wait
  } States;

  States state, next;

  always_comb begin
    case (state)
      S: next = States'(data ? S1: S);
      S1: next = States'(data ? S11: S);
      S11: next = States'(data ? S11 : S110);
      S110: next = States'(data ? B0 : S);
      B0: next = B1;
      B1: next = B2;
      B2: next = B3;
      B3: next = Count;
      Count: next = States'(done_counting ? Wait : Count);
      Wait: next = States'(ack ? S : Wait);
      default: next = States'(4'bx);
    endcase
  end

  always @(posedge clk) begin
    if (reset) state <= S;
    else state <= next;
  end

  always_comb begin
    shift_ena = 0; counting = 0; done = 0;
    if (state == B0 || state == B1 || state == B2 || state == B3)
      shift_ena = 1;
    if (state == Count)
      counting = 1;
    if (state == Wait)
      done = 1;

    if (|state === 1'bx) begin
      {shift_ena, counting, done} = 'x;
    end

  end

endmodule
