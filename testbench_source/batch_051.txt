module shift_register (
  input clk,
  input reset,
  input enable,
  input d,
  output reg q
);

  reg [2:0] shift_reg;

  always @(posedge clk) begin
    if (reset) begin
      shift_reg <= 3'b0;
      q <= 1'b0;
    end else if (enable) begin
      shift_reg <= {shift_reg[1:0], d};
      q <= shift_reg[2];
    end
  end

endmodule

module twos_complement (
    input [3:0] a,
    output reg [3:0] twos_comp
);

reg [3:0] ones_comp;
reg [3:0] inverted;
reg [3:0] add_one;

always @ (a) begin
    ones_comp = ~a;
    inverted = ones_comp + 4'b1;
    add_one = inverted[3] ? inverted + 4'b1 : inverted;
    twos_comp <= add_one;
end

endmodule

module priority_mux (
    input [3:0] in,
    output reg out
);

reg [1:0] stage1_out;
reg [1:0] stage2_out;

always @ (in) begin
    stage1_out[0] <= in[0] | in[2];
    stage1_out[1] <= in[1] | in[3];
end

always @ (stage1_out) begin
    stage2_out[0] <= stage1_out[0] | stage1_out[1];
    stage2_out[1] <= stage1_out[0] & stage1_out[1];
end

always @ (stage2_out) begin
    if (stage2_out[1]) begin
        out <= in[3];
    end else if (stage2_out[0]) begin
        out <= in[2];
    end else if (stage1_out[1]) begin
        out <= in[1];
    end else begin
        out <= in[0];
    end
end

endmodule
