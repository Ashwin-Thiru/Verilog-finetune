);

    Q_reg1 <= Q;
end

always @(posedge clk) begin
    Q_reg2 <= Q_reg1;

always @(posedge clk) begin
    Q_reg3 <= Q_reg2;
end

always @(posedge clk) begin
    if (Q_reg3 == 4'b1111) begin
        Q <= 4'b0000;

module accu(
    input               clk         ,   
    input               rst_n       ,
    input       [7:0]   data_in     ,
    input               valid_a     ,
    input               ready_b     ,
 
    output              ready_a     ,
    output  reg         valid_b     ,
    output  reg [9:0]   data_out
);

reg [7:0] data_reg;
reg [9:0] acc_reg;

always @(posedge clk or negedge rst_n) begin
    if(!rst_n) begin
        data_reg <= 8'b0;
        acc_reg <= 10'b0;
        valid_b <= 1'b0;
    end
    else begin
        data_reg <= valid_a ? data_in : data_reg;
        acc_reg <= acc_reg + data_reg;
        valid_b <= valid_a;
    end
end

assign ready_a = !valid_b || ready_b;

always @(posedge clk or negedge rst_n) begin
    if(!rst_n) begin
        data_out <= 10'b0;
    end
    else if(valid_b && ready_a) begin
        data_out <= acc_reg;
    end
end

endmodule
