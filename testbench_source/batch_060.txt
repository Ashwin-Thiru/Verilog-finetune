module mux6to1_pipeline (
    input [2:0] sel,
    input [3:0] data0,
    input [3:0] data1,
    input [3:0] data2,
    input [3:0] data3,
    input [3:0] data4,
    input [3:0] data5,
    output reg [3:0] out
);

reg [3:0] stage1_out [5:0];
reg [3:0] stage2_out [2:0];

always @(*) begin
    case(sel)
        3'b000: begin
            stage1_out[0] = data0 & 4'h0F;
            stage1_out[1] = data1 & 4'h0F;
            stage1_out[2] = data2 & 4'h0F;
            stage1_out[3] = data3 & 4'h0F;
            stage1_out[4] = data4 & 4'h0F;
            stage1_out[5] = data5 & 4'h0F;
        end
        3'b001: begin
            stage1_out[0] = data0 | 4'hF0;
            stage1_out[1] = data1 | 4'hF0;
            stage1_out[2] = data2 | 4'hF0;
            stage1_out[3] = data3 | 4'hF0;
            stage1_out[4] = data4 | 4'hF0;
            stage1_out[5] = data5 | 4'hF0;
        end
        3'b010: begin
            stage1_out[0] = data0 & 4'h0F;
            stage1_out[1] = data1 & 4'h0F;
            stage1_out[2] = data2 & 4'h0F;
            stage1_out[3] = data3 & 4'h0F;
            stage1_out[4] = data4 & 4'h0F;
            stage1_out[5] = data5 & 4'h0F;
        end
        3'b011: begin
            stage1_out[0] = data0 | 4'hF0;
            stage1_out[1] = data1 | 4'hF0;
            stage1_out[2] = data2 | 4'hF0;
            stage1_out[3] = data3 | 4'hF0;
            stage1_out[4] = data4 | 4'hF0;
            stage1_out[5] = data5 | 4'hF0;
        end
        3'b100: begin
            stage1_out[0] = 4'h0;
            stage1_out[1] = 4'h0;
            stage1_out[2] = 4'h0;
            stage1_out[3] = 4'h0;
            stage1_out[4] = 4'h0;
            stage1_out[5] = 4'h0;
        end
        default: begin
            stage1_out[0] = 4'h0;
            stage1_out[1] = 4'h0;
            stage1_out[2] = 4'h0;
            stage1_out[3] = 4'h0;
            stage1_out[4] = 4'h0;
            stage1_out[5] = 4'h0;
        end
    endcase
end

always @(*) begin
    stage2_out[0] = stage1_out[0] | stage1_out[1];
    stage2_out[1] = stage1_out[2] | stage1_out[3];
    stage2_out[2] = stage1_out[4] | stage1_out[5];
end

always @(*) begin
    out = 3'b000;
    case(sel)
        3'b000: out = stage2_out[0];
        3'b001: out = stage2_out[0];
        3'b010: out = stage2_out[1];
        3'b011: out = stage2_out[1];
        3'b100: out = stage2_out[2];
        default: out = 3'b000;
    endcase
end

endmodule

    input [7:0] b,
    output reg [15:0] result
);

            a_reg = a;
